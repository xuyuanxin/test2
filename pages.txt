

#$ make -f Makefile_3_88

.PHONY: all clean

MKDIR   = mkdir
RM      = rm
RMFLAGS = -fr

CC = gcc
ROOT = $(realpath ..)
DIR_OBJS = $(ROOT)/build/objs
#DIR_OBJS = objs
DIR_EXES = exes

DIRS = $(DIR_OBJS) $(DIR_EXES)

EXE = complicated.exe
EXE := $(addprefix $(DIR_EXES)/, $(EXE))

SRCS = $(wildcard *.c)
OBJS = $(SRCS:.c=.o)
OBJS := $(addprefix $(DIR_OBJS)/, $(OBJS))

all: $(DIRS) $(EXE)
#all:  $(EXE)
$(DIRS):
	$(MKDIR) $@
$(EXE): $(OBJS)
	$(CC) -o $@ $^
$(DIR_OBJS)/%.o: %.c
	$(CC) -o $@ -c $^

#$ make -f Makefile_3_88 clean
clean:
	$(RM) $(RMFLAGS) $(DIRS) $(EXE)
